  if (mode === "over")  return rolledNumber > target;
  if (mode === "under") return rolledNumber < target;
  return false;
}

// Multiplier formula (house edge 1%)
function calcMultiplier(winChance) {
  return parseFloat(((99 / winChance)).toFixed(4));
}

// Win chance
function calcWinChance(target, mode) {
  if (mode === "over")  return parseFloat((100 - target).toFixed(2));
  if (mode === "under") return parseFloat((target - 1).toFixed(2));
}
