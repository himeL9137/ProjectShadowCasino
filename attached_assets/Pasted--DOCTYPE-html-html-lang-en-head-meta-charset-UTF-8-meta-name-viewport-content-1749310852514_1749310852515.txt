    <div class="container">
        <h1>Plinko Game</h1>
        <input type="number" id="betAmount" placeholder="Enter bet amount (৳)" min="0.01" step="0.01">
        <button onclick="dropBall()">Drop Ball</button>
        <div id="result"></div>
        <div class="multiplier-bar" id="multiplierBar"></div>
    </div>

    <script type="text/typescript">
        // Define the multiplier bar
        const multipliers: number[] = [2.0, 1.8, 1.6, 1.4, 1.0, 0.8, 0.6, 0.4, 0.4, 0.6, 0.8, 1.0, 1.4, 1.6, 1.8];

        // Interface for game result
        interface PlinkoResult {
            slotIndex: number;
            multiplier: number;
            payout: number;
            outcome: 'Win' | 'Loss' | 'Break-even';
        }

        // Function to simulate ball drop and calculate result
        function dropBall(): void {
            const betInput = document.getElementById('betAmount') as HTMLInputElement;
            const betAmount = parseFloat(betInput.value);
            const resultDiv = document.getElementById('result')!;
            const multiplierBar = document.getElementById('multiplierBar')!;

            // Clear previous highlights
            document.querySelectorAll('.multiplier').forEach(el => el.classList.remove('active'));

            // Validate bet amount
            if (isNaN(betAmount) || betAmount <= 0) {
                resultDiv.innerHTML = '<span class="error">Please enter a valid bet amount!</span>';
                return;
            }

            // Simulate ball drop by selecting a slot index
            const slotIndex = Math.floor(Math.random() * multipliers.length);
            const selectedMultiplier = multipliers[slotIndex];

            // Calculate payout
            const payout = betAmount * selectedMultiplier;

            // Determine outcome
            let outcome: 'Win' | 'Loss' | 'Break-even';
            if (payout > betAmount) {
                outcome = 'Win';
            } else if (payout < betAmount) {
                outcome = 'Loss';
            } else {
                outcome = 'Break-even';
            }

            // Highlight selected multiplier
            const multiplierElements = document.querySelectorAll('.multiplier');
            if (multiplierElements[slotIndex]) {
                multiplierElements[slotIndex].classList.add('active');
            }

            // Display result with Bangladeshi Taka (৳)
            const result: PlinkoResult = { slotIndex, multiplier: selectedMultiplier, payout, outcome };
            resultDiv.innerHTML = `
                Slot: ${result.slotIndex + 1} (Multiplier: ${result.multiplier}x)<br>
                Payout: ৳${result.payout.toFixed(2)}<br>
                Outcome: <span class="${result.outcome.toLowerCase()}">${result.outcome}</span>
            `;
        }

        // Display multiplier bar on load
        window.onload = () => {
            const bar = document.getElementById('multiplierBar')!;
            multipliers.forEach(multiplier => {
                const div = document.createElement('div');
                div.className = 'multiplier';
                div.textContent = `${multiplier}x`;
                bar.appendChild(div);
            });

            // Add event listener for Enter key
            const betInput = document.getElementById('betAmount') as HTMLInputElement;
            betInput.addEventListener('keypress', (e) => {
                if (e.key === 'Enter') {
                    dropBall();
                }
            });
        };
    </script>
