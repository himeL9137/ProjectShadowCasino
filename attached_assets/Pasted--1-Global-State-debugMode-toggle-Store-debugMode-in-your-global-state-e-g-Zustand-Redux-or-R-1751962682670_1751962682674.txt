// components/admin/DebugToggle.tsx
import { useDebugStore } from "@/store/debugStore"
import { useSession } from "next-auth/react" // Or your auth method

export default function DebugToggle() {
  const { debugMode, setDebugMode } = useDebugStore()
  const { data: session } = useSession()

  if (session?.user?.username !== "shadowHimel") return null

  return (
    <div className="p-4 border rounded bg-card shadow-md">
      <h2 className="text-xl font-semibold mb-2">Debug Mode</h2>
      <label className="flex items-center gap-3">
        <input
          type="checkbox"
          checked={debugMode}
          onChange={(e) => setDebugMode(e.target.checked)}
        />
        <span>{debugMode ? "ON" : "OFF"}</span>
      </label>
    </div>
  )
}

export function conditionalClass(base: string, debug: boolean, fancy: string) {
  return debug ? base : `${base} ${fancy}`
}
