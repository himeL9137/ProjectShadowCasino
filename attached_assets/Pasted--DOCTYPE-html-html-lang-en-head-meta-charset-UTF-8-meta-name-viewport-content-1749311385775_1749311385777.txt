<body>
    <div class="container">
        <h1>Plinko Master</h1>
        <input type="number" id="betAmount" placeholder="Enter bet amount (৳)" min="0.01" step="0.01">
        <button onclick="playPlinko()">Drop Ball</button>
        <div id="result"></div>

            <div id="ball" class="ball"></div>
        </div>
        <div class="multiplier-bar" id="multiplierBar"></div>
    </div>

    <script type="text/typescript">
        // Multiplier bar (for client-side display)
        const multipliers: number[] = [2.0, 1.8, 1.6, 1.4, 1.0, 0.8, 0.6, 0.4, 0.4, 0.6, 0.8, 1.0, 1.4, 1.6, 1.8];

        interface PlinkoResult {
            slotIndex: number;
            multiplier: number;
            payout: number;
            outcome: 'Win' | 'Loss' | 'Break-even';
            newBalance: number;
        }

        // WebSocket connection
        const ws = new WebSocket('wss://' + window.location.host);
        ws.onopen = () => console.log('WebSocket connected');
        ws.onerror = (error) => console.error('WebSocket error:', error);
        ws.onclose = () => console.log('WebSocket disconnected');

        ws.onmessage = (event) => {
            const data = JSON.parse(event.data);
            if (data.type === 'PLINKO_RESULT') {
                displayResult(data.result);
            }
        };

        // Function to animate ball drop and display result
        function displayResult(result: PlinkoResult): void {
            const resultDiv = document.getElementById('result')!;
            const multiplierBar = document.getElementById('multiplierBar')!;
            const ball = document.getElementById('ball')!;

            // Clear previous highlights
            document.querySelectorAll('.multiplier').forEach(el => el.classList.remove('active'));

            // Animate ball to the selected slot
            const slotWidth = multiplierBar.clientWidth / multipliers.length;
            const slotX = result.slotIndex * slotWidth + slotWidth / 2 - 10; // Center ball in slot
            ball.style.top = '90%'; // Move to bottom of board
            ball.style.left = `${slotX}px`;

            // Highlight selected multiplier
            const multiplierElements = document.querySelectorAll('.multiplier');
            if (multiplierElements[result.slotIndex]) {
                multiplierElements[result.slotIndex].classList.add('active');
            }

            // Display result
            resultDiv.innerHTML = `
                Slot: ${result.slotIndex + 1} (Multiplier: ${result.multiplier}x)<br>
                Payout: ৳${result.payout.toFixed(2)}<br>
                Outcome: <span class="${result.outcome.toLowerCase()}">${result.outcome}</span><br>
                New Balance: ৳${result.newBalance.toFixed(2)}
            `;
        }

        // Function to play Plinko
        async function playPlinko(): Promise<void> {
            const betInput = document.getElementById('betAmount') as HTMLInputElement;
            const betAmount = parseFloat(betInput.value);
            const resultDiv = document.getElementById('result')!;
            const ball = document.getElementById('ball')!;

            // Reset ball position
            ball.style.top = '0';
            ball.style.left = '50%';

            // Validate bet amount
            if (isNaN(betAmount) || betAmount <= 0) {
                resultDiv.innerHTML = '<span class="error">Please enter a valid bet amount!</span>';
                return;
            }

            try {
                const response = await fetch('/api/games/play', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Authorization': `Bearer ${document.cookie.split('; ').find(row => row.startsWith('jwt='))?.split('=')[1] || ''}`,
                    },
                    body: JSON.stringify({
                        gameType: 'PLINKO_MASTER',
                        betAmount,
                        currency: 'BDT',
                        numBalls: 1,
                        forceWin: false,
                        forceLose: false,
                    }),
                });

                if (!response.ok) {
                    const error = await response.json();
                    resultDiv.innerHTML = `<span class="error">Error: ${error.message}</span>`;
                    return;
                }

                // Result is handled via WebSocket
            } catch (error) {
                resultDiv.innerHTML = `<span class="error">Network error: ${error}</span>`;
            }
        }

        // Display multiplier bar on load
        window.onload = () => {
            const bar = document.getElementById('multiplierBar')!;
            multipliers.forEach(multiplier => {
                const div = document.createElement('div');
                div.className = 'multiplier';
                div.textContent = `${multiplier}x`;
                bar.appendChild(div);
            });

            // Add event listener for Enter key
            const betInput = document.getElementById('betAmount') as HTMLInputElement;
            betInput.addEventListener('keypress', (e) => {
                if (e.key === 'Enter') {
                    playPlinko();
                }
            });
        };
    </script>

    </script>
