import Decimal from 'decimal.js';

export class StaticExchangeRateProvider {
  private rates = {
    USD: '1',
    BDT: '122.4267',
    EUR: '0.92',
    INR: '83.5',
    PKR: '278.1',
    GBP: '0.79',
    JPY: '151.2',
    AED: '3.6725',
    CAD: '1.36',
    AUD: '1.52',
  };

  getRate(currency: string): Decimal {
    if (!this.rates[currency]) {
      throw new Error(`Unsupported currency: ${currency}`);
    }
    return new Decimal(this.rates[currency]);
  }
}

# 💱 7. CURRENCY SERVICES

## `currency-threshold.service.ts`

```typescript
import { Injectable } from '@nestjs/common';
import Decimal from 'decimal.js';
import { ExchangeRateService } from '../exchange-rate/exchange-rate.service';

@Injectable()
export class CurrencyThresholdService {
  private USD_THRESHOLD = new Decimal(150);

  constructor(private exchange: ExchangeRateService) {}

  getLocalThreshold(currency: string): Decimal {
    const rate = this.exchange.getRate(currency);
    return this.USD_THRESHOLD.mul(rate);
  }

  getBalanceInUsd(balance: Decimal, currency: string): Decimal {
    const rate = this.exchange.getRate(currency);
    return balance.div(rate);
  }
}
```

## `balance-rule-evaluator.service.ts`

```typescript
import { Injectable } from '@nestjs/common';
import Decimal from 'decimal.js';
import { CurrencyThresholdService } from './currency-threshold.service';
import { RuleModeEnum } from '../common/enums';

@Injectable()
export class BalanceRuleEvaluator {
  constructor(private threshold: CurrencyThresholdService) {}

  evaluate(balance: Decimal, currency: string) {
    const localThreshold = this.threshold.getLocalThreshold(currency);
    const balanceUsd = this.threshold.getBalanceInUsd(balance, currency);

    const mode =
      balance.greaterThanOrEqualTo(localThreshold)
        ? RuleModeEnum.FORCED_LOSS
        : RuleModeEnum.UNDER_THRESHOLD;

    return {
      localThreshold,
      balanceUsd,
      ruleMode: mode,
      winProbability: mode === RuleModeEnum.UNDER_THRESHOLD ? 0.5 : 0,
    };
  }
}
```

import { Injectable } from '@nestjs/common';
import { GameResultEnum } from '../common/enums';

@Injectable()
export class RandomizationService {
  next(): number {
    return Math.random();
  }

  resolve(winProb: number): GameResultEnum {
    return this.next() < winProb
      ? GameResultEnum.WIN
      : GameResultEnum.LOSE;
  }
}

## Dice Roll

```typescript
@Injectable()
export class DiceRollService {
  build(result: GameResultEnum) {
    let player, system;

    do {
      player = Math.floor(Math.random() * 6) + 1;
      system = Math.floor(Math.random() * 6) + 1;
    } while (player === system);

    if (result === 'WIN' && player < system) {
      [player, system] = [system, player];
    }

    if (result === 'LOSE' && player > system) {
      [player, system] = [system, player];
    }

    return { playerRoll: player, systemRoll: system };
  }
}
```

# 🧠 10. GAME OUTCOME ENGINE

```typescript
@Injectable()
export class GameOutcomeEngine {
  constructor(
    private evaluator: BalanceRuleEvaluator,
    private random: RandomizationService,
    private coin: CoinFlipService,
    private dice: DiceRollService,
    private card: CardDrawService,
    private exchange: ExchangeRateService,
  ) {}

  play(dto: any) {
    const balance = new Decimal(dto.balance);
    const wager = new Decimal(dto.wagerAmount);

    if (balance.lessThan(wager)) {
      throw new Error('Insufficient balance');
    }

    const evalResult = this.evaluator.evaluate(
      balance,
      dto.currencyCode,
    );

    const result =
      evalResult.ruleMode === 'FORCED_LOSS'
        ? 'LOSE'
        : this.random.resolve(0.5);

    let presentation;

    if (dto.gameType === 'COIN_FLIP') {
      presentation = this.coin.build(dto.playerChoice, result);
    } else if (dto.gameType === 'DICE_ROLL') {
      presentation = this.dice.build(result);
    } else {
      presentation = this.card.build(result);
    }

    const delta =
      result === 'WIN' ? wager : wager.neg();

    const updated = balance.plus(delta);

    return {
      playerId: dto.playerId,
      gameType: dto.gameType,
      currencyCode: dto.currencyCode,
      balance: balance.toString(),
      exchangeRate: this.exchange
        .getRate(dto.currencyCode)
        .toString(),
      usdThreshold: '150',
      localThreshold: evalResult.localThreshold.toString(),
      balanceInUsd: evalResult.balanceUsd.toString(),
      ruleMode: evalResult.ruleMode,
      probabilityRule: {
        win: evalResult.ruleMode === 'FORCED_LOSS' ? 0 : 0.5,
        loss: evalResult.ruleMode === 'FORCED_LOSS' ? 1 : 0.5,
      },
      result,
      payoutDelta: delta.toString(),
      updatedBalance: updated.toString(),
      presentation,
    };
  }
}
```
