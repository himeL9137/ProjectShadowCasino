    <div class="container">
        <h1>Plinko Game</h1>
        <input type="number" id="betAmount" placeholder="Enter bet amount" min="0.01" step="0.01">
        <button onclick="dropBall()">Drop Ball</button>
        <div id="result"></div>
        <div class="multiplier-bar" id="multiplierBar"></div>
    </div>

    <script type="text/typescript">
        // Define the multiplier bar
        const multipliers: number[] = [2.0, 1.8, 1.6, 1.4, 1.0, 0.8, 0.6, 0.4, 0.4, 0.6, 0.8, 1.0, 1.4, 1.6, 1.8];

        // Interface for game result
        interface PlinkoResult {
            multiplier: number;
            payout: number;
            outcome: string;
        }

        // Function to simulate ball drop and calculate result
        function dropBall(): void {
            const betInput = document.getElementById('betAmount') as HTMLInputElement;
            const betAmount = parseFloat(betInput.value);

            // Validate bet amount
            if (isNaN(betAmount) || betAmount <= 0) {
                document.getElementById('result')!.innerHTML = 'Please enter a valid bet amount!';
                return;
            }

            // Randomly select a multiplier
            const selectedMultiplier = multipliers[Math.floor(Math.random() * multipliers.length)];
            
            // Calculate payout
            const payout = betAmount * selectedMultiplier;

            // Determine outcome
            let outcome: string;
            if (payout > betAmount) {
                outcome = 'Win';
            } else if (payout < betAmount) {
                outcome = 'Loss';
            } else {
                outcome = 'Break-even';
            }

            // Display result
            const result: PlinkoResult = { multiplier: selectedMultiplier, payout, outcome };
            document.getElementById('result')!.innerHTML = `
                Multiplier: ${result.multiplier}x<br>
                Payout: $${result.payout.toFixed(2)}<br>
                Outcome: ${result.outcome}
            `;
        }

        // Display multiplier bar on load
        window.onload = () => {
            const bar = document.getElementById('multiplierBar')!;
            multipliers.forEach(multiplier => {
                const div = document.createElement('div');
                div.className = 'multiplier';
                div.textContent = `${multiplier}x`;
                bar.appendChild(div);
            });
        };
    </script>

    <script>

        const script = document.querySelector('script[type="text/typescript"]').textContent;

    </script>
